// ----- Инициализация ----- let balance = Number(localStorage.getItem('balance')) || 100; let isRacing = false; let firstBallTouched = false; const TOTAL_BALLS = 66; const WORLD_HEIGHT = 3000; function updateBal() { document.getElementById('bal').textContent = balance.toFixed(2); localStorage.setItem('balance', balance); } // ----- Создаём движок и рендер ----- const engine = Matter.Engine.create(); engine.gravity.y = 1.2; const w = window.innerWidth; const h = window.innerHeight; const render = Matter.Render.create({ element: document.getElementById('canvas-container'), engine: engine, options: { width: w, height: h, wireframes: false, background: '#0a0e27', hasBounds: true } }); // ----- Создаём стены и препятствия ----- const walls = [ Matter.Bodies.rectangle(w/2, WORLD_HEIGHT + 50, w*2, 100, { isStatic: true, render: {fillStyle:'#00ff00'} }), Matter.Bodies.rectangle(-20, WORLD_HEIGHT/2, 40, WORLD_HEIGHT, { isStatic: true }), Matter.Bodies.rectangle(w+20, WORLD_HEIGHT/2, 40, WORLD_HEIGHT, { isStatic: true }) ]; const pins = []; for(let l=0; l<40; l++){ let rowPins = (l % 2 === 0) ? 8 : 7; let spacing = w / rowPins; for(let i=0; i<=rowPins; i++){ pins.push(Matter.Bodies.circle(i*spacing + (l%2?spacing/2:0), 400 + l*65, 6, { isStatic:true, render:{fillStyle:'#444'}, restitution:1.2 })); } } // ----- Финишные зоны ----- const zoneData = [ { x: w*0.15, label:'0x', color:'#ff4444', mult:0 }, { x: w*0.40, label:'1x', color:'#0066ff', mult:1 }, { x: w*0.65, label:'2x', color:'#ffbb00', mult:2 }, { x: w*0.90, label:'30x', color:'#ff00ff', mult:30 } ]; zoneData.forEach(z=>{ const sensor = Matter.Bodies.rectangle(z.x, WORLD_HEIGHT-20, w*0.22, 60, { isStatic:true, isSensor:true, render:{fillStyle: z.color}, label:'ZONE_'+z.mult }); Matter.Composite.add(engine.world, sensor); }); Matter.Composite.add(engine.world, [...walls, ...pins]); Matter.Render.run(render); Matter.Runner.run(Matter.Runner.create(), engine); // ----- Запуск игры ----- function startGame(bet){ if(balance < bet || isRacing) return; isRacing = true; firstBallTouched = false; balance -= bet; updateBal(); document.getElementById('winPopup').style.display='none'; document.querySelectorAll('.btn').forEach(b=>b.disabled=true); // Очистка старых шаров Matter.Composite.allBodies(engine.world).forEach(b=>{ if(!b.isStatic) Matter.Composite.remove(engine.world, b); }); // Высыпаем шары с небольшим интервалом for(let i=0; i{ let m, col, txt; if(i<50){ m=0; col='#ff4444'; txt='0x'; } else if(i<60){ m=1; col='#0066ff'; txt='1x'; } else if(i<65){ m=2; col='#ffbb00'; txt='2x'; } else{ m=30; col='#ff00ff'; txt='30x'; } const ball = Matter.Bodies.circle(w/2 + (Math.random()*60-30), 100, 14, { restitution:0.6, friction:0.001, render:{fillStyle: col}, label:'BALL', customMult: m, customTxt: txt, betAmount: bet }); Matter.Composite.add(engine.world, ball); }, i*30); } } // ----- Камера и отрисовка текста ----- Matter.Events.on(render, 'afterRender', ()=>{ const context = render.context; const balls = Matter.Composite.allBodies(engine.world).filter(b=>b.label==='BALL'); if(balls.length>0 && isRacing){ // Центрируем камеру по самому нижнему шару let lowestY = Math.max(...balls.map(b=>b.position.y)); let targetY = Math.min(Math.max(lowestY - h/2, 0), WORLD_HEIGHT - h); Matter.Render.lookAt(render, {min:{x:0,y:targetY}, max:{x:w,y:targetY+h}}); } // Рисуем текст на шарах context.font="bold 12px Arial"; context.textAlign="center"; context.fillStyle="white"; balls.forEach(b=>{ context.fillText(b.customTxt, b.position.x, b.position.y + 5); }); }); // ----- Обработка выигрыша ----- Matter.Events.on(engine, 'collisionStart', event=>{ if(firstBallTouched) return; event.pairs.forEach(pair=>{ const ball = [pair.bodyA, pair.bodyB].find(b=>b.label==='BALL'); const zone = [pair.bodyA, pair.bodyB].find(b=>b.label.startsWith('ZONE_')); if(ball && zone){ firstBallTouched=true; const mult = parseInt(zone.label.split('_')[1]); handleWin(mult, ball.betAmount); } }); }); function handleWin(mult, bet){ const winAmount = bet * mult; balance += winAmount; updateBal(); const pop = document.getElementById('winPopup'); pop.style.display='block'; pop.style.color = mult===0?'#ff4444':(mult===30?'#ff00ff':'#00ff00'); pop.innerHTML = mult>0?`ПЕРВЫМ ПРИШЕЛ ${mult}x!
ВЫИГРЫШ: ${winAmount} TON`:`ПЕРВЫМ ПРИШЕЛ 0x...
СТАВКА ПРОИГРАНА`; if(window.Telegram?.WebApp) window.Telegram.WebApp.HapticFeedback.notificationOccurred(mult>0?'success':'error'); setTimeout(()=>{ isRacing=false; document.querySelectorAll('.btn').forEach(b=>b.disabled=false); }, 2000); }
